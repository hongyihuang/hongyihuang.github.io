def set_latex_power(x_var, x_name, x_unit, power):
    '''Adds power to variable name, conventional name and unit name.
    '''
    return (x_var + "^{" + str(power) + "}", \
            x_name + "^{" + str(power) + "}",\
            "("+x_unit + ")^{" + str(power) + "}")

def get_latex_power(name):
    '''Gets power of a name
    '''
    ans = name.split('^')
    if (len(ans)==1):
        return 1
    else:
        return ans[1]
def axislabel(name, unit):
    return '  ${}({})$  '.format(name, unit)
def theta_str(theta):
    theta=np.round(theta, decimals=3)
    ans = "$y=" + str(theta[0]) + "x^" + str(len(theta)-1)
    theta=theta[1:]
    for theta_i, i in zip(theta, range(0, len(theta))):
        if (theta_i<0.0):
            ans+=str(theta_i)
        else:
            ans+= "+" + str(theta_i)
        ans+= "x^" + str(len(theta)-1-i)
    ans+= "$"
    return ans
def plotdata(x, y, x_var, y_var, x_name, y_name, x_unit, y_unit, power):
    xp = np.linspace(0, max(x)*1.1, 100)
    
    fig = plt.figure(figsize=(10, 7))
    ax = fig.gca()
    ax.set_title("${}$ vs ${}$".format(y_name, x_name), fontsize=25) # title of plot
    ax.set_xlabel(axislabel(x_name, x_unit), fontsize=15) #xlabel
    ax.set_ylabel(axislabel(y_name, y_unit), fontsize=15) #ylabel

    datapoints = ax.plot(x, y, 'o', label="Datapoints")
    ax.grid()
    ax.set_axisbelow(True)
    ax.set_xlim([0, max(x)*1.1])
    ax.set_ylim([0, max(y)*1.1])
    
    return fig

def tryfit(x, y, x_var, y_var, x_name, y_name, x_unit, y_unit, power):
    xp = np.linspace(0, max(x)*1.1, 100)
    
    theta, res, _, _, _  = np.polyfit(x, y, power, full=True)
    feedforward = np.poly1d(theta)
    
    res = np.round(1-res[0], 3)
    fig = plt.figure(figsize=(10, 7))
    ax = fig.gca()
    ax.set_title("${}$ vs ${}$".format(y_name, x_name), fontsize=25) # title of plot
    ax.set_xlabel(axislabel(x_name, x_unit), fontsize=15) #xlabel
    ax.set_ylabel(axislabel(y_name, y_unit), fontsize=15) #ylabel
    datapoints = ax.plot(x, y, 'o', label="Datapoints")
    regression = ax.plot(xp, feedforward(xp), '-', label=theta_str(theta) + " $R^2 = {}$".format(res))
    ax.grid()
    ax.legend()
    ax.set_axisbelow(True)
    ax.set_xlim([0, max(x)*1.1])
    ax.set_ylim([0, max(y)*1.1])
    fig.savefig('.png'.format(y_var, x_var))
    
    return fig

def linearfit(x, y, x_var, y_var, x_name, y_name, x_unit, y_unit):
    xp = np.linspace(0, max(x)*1.1, 100)

    x = x[:,np.newaxis]
    theta, res, _, _ = np.linalg.lstsq(x, y)
    
    res = np.round(1-res[0], 3)
    fig = plt.figure(figsize=(10, 7))
    ax = fig.gca()
    ax.set_title("${}$ vs ${}$".format(y_name, x_name), fontsize=25) # title of plot
    ax.set_xlabel(axislabel(x_name, x_unit), fontsize=15) #xlabel
    ax.set_ylabel(axislabel(y_name, y_unit), fontsize=15) #ylabel
    datapoints = ax.plot(x, y, 'o', label="Datapoints")
    regression = ax.plot(xp, theta*xp, '-', label=theta_str(np.append(theta, [0])) + " $R^2 = {}$".format(res))
    ax.grid()
    ax.legend()
    ax.set_axisbelow(True)
    ax.set_xlim([0, max(x)*1.1])
    ax.set_ylim([0, max(y)*1.1])
    
    return fig
    
def hypothesis(x, y, x_var, y_var, x_name, y_name, x_unit, y_unit, power, originalfit, zero):
    hx_var, hx_name, hx_unit = set_latex_power(x_var, x_name, x_unit, power)
    t_title = [axislabel(x_name, x_unit), \
               axislabel(hx_name, hx_unit), \
               axislabel(y_name, y_unit)]
    hx = np.round(np.power(x, power), decimals=2)
    t_data = np.array([x, hx, y]).T
    t_df = pd.DataFrame(t_data, columns=t_title)
    if originalfit:
        dataplot = tryfit(x, y, x_var, y_var, x_name, y_name, x_unit, y_unit, power)
    else:
        dataplot = plotdata(x, y, x_var, y_var, x_name, y_name, x_unit, y_unit, power)
    if zero:
        fitplot = linearfit(hx, y, hx_var, y_var, hx_name, y_name, hx_unit, y_unit)
    else:
        fitplot = tryfit(hx, y, hx_var, y_var, hx_name, y_name, hx_unit, y_unit, 1)
    return t_df, dataplot, fitplot

  2. The data does support the square root relationship since the square rooted time leads to a linearized data trend that has a mean square error close to 1. Concidering a +/- 0.1 m measurement inaccuracy, the hypothesis is sufficiently correct. Value for constant a is as follows:
    $x=\sqrt{a}\sqrt{t}$
    $y=2.054x$
    $\sqrt{a}=2.0747$ $a=2.0747^2=4.304$ (acceleration can only be positive)

  3. The percent uncertainties are:

    • 1.4m: $\frac{Probable Error}{Measurment}\times100\% = \frac{0.1m}{1.4m}\times100\% = 7.14\%$
    • 3.0m: $\frac{Probable Error}{Measurment}\times100\% = \frac{0.1m}{3.0m}\times100\% = 3.33\%$
    • Therefore, the 3.0m measurement is more reliable as the % uncertainty is lower than that 1.4m measurment.
